What is Function in Javascript ?

# A function is a piece of block of code which does not execute untill & unless it is called.

Function declaration
Javascript 

function myName() {
    console.log('my name is avinash kumar');
}

myName(); // my name is avinash kumar

# function parameter.

Javascript 

function addNums(num1, num2) {
    console.log(num1 + num2);
}

addNums(5, 4); // 9 
addNums(); // result will be NaN means not a number
addNums(10); // NaN

NOTE - Result is NaN because we have not passes the arguments value to the function which is required. Even if we pass (1) one argument Result will be NaN

Javascript 

let myAge = 22;

function addsToMyAge(IncrementBy) {
    myAge = myAge + IncrementBy;
    console.log(myAge);
}
addsToMyAge(10); // 32

# Function to return the passing number is even or not.

Javascript 

function isEven(number) {
    if (number % 2 === 0) {
        return true;
    } else {
        return false;
    }
}

console.log(isEven(5)); // false 
console.log(isEven(10)); // true

# Function to return first character of passing anyString to the parameter.

Javascript 

function firstChar(anyString) {
    return anyString[0];
}

console.log(firstChar('Avinash')); // A

# Function to find index of target, if it is not found then return -1.

Javascript 

function findTarget(array, target) {
    for (let i = 0; i < array.length; i++) {
        if (array[i] === target) {
            return i;
        }
    }
    return -1;
}

const myArray = [1, 3, 8, 9, 100];
const answer1 = findTarget(myArray, 100);
const answer2 = findTarget(myArray, 50);

console.log(answer1, answer2); // 4 -1
Passing default value to the parameter

# old Method

Javascript 

function addTwo(a,b) {
    if(typeof b === "undefined"){
        b = 1;
    }
    return a + b;
}

const ans = addTwo(4);
console.log(ans);     // 5
Javascript 

// Here we are passing the default value to parameter 

function addNums(num1 = 1, num2 = 1) {
    console.log(num1 + num2);
}

addNums(); // 2-- taking the default value 
addNums(5, 5); // 10
Function Expression

# We simply stored a function in a variable and we use variable name to call this function.

Javascript 

const isEven = function (number) {
    return number % 2 === 0;
}

console.log(isEven(2)); // true
Function return
Javascript

function addNums(num1, num2) {
    //console.log(num1 + num2) 
    return num1 + num2;
}

console.log(addNums(5, 5)); // 10

Note - Any statement after return never execute.

Nested Function

# see the Arrow Function section before this example.

Javascript 

function app() {
    const myFunc = () => {
        console.log('hello from my func..');
    }

    const addTwo = (num1, num2) => {
        return num1 + num2;
    }

    const mult = (num1, num2) => num1 * num2;

    console.log('inside app..');
    myFunc();
    console.log(addTwo(5, 10));
    console.log(mult(5, 10));
}

app(); 
// inside app.. 
// hello from my func.. 
// 15 
// 50
