How to iterate on Object in JS ?

# We can iterate on object

1. using For in loop
2. using Object.keys

Javascript 

const person = {
    name: "avinash",
    age: 20,
    hobbies: ["coding", "music", "sports"]
}
Javascript 

for (let key in person) {
    console.log(key);
}

// name 
// age 
// hobbies
Javascript 

for (let key in person) {
    console.log(person[key]);
};

// avinash 
// 20 
// (3) ['coding', 'music', 'sports']
Javascript 

for (let key in person) {
    console.log(`$ {key}: $ {person[key]}`);
    // using template string we will get key & value pair both
}

// name: avinash 
// age:20 
// hobbies:coding, music, sports
Javascript 

for (let key in person) {
    console.log(key, ".", person[key]);
    // we will get same result as above
};

// name: avinash 
// age: 20 
// hobbies : (3) ['coding', 'music', 'sports']
Object.keys method
Javascript 

// This return arrays of keys starting index from 0

console.log(Object.keys(person));
// ['name', 'age', 'hobbies']

NOTE - We can use For of loop also instead of For in.

computed properties
Javascript 

const key1 = "objkey1";
const key2 = "objkey2";

const value1 = "myvalue1";
const value2 = "myvalue2";

Javascript

    [key1]: value1,
    [key2]: value2
};

// console.log(obj); 
// {objkey1: 'myvalue1', objkey2: 'myvalue2"}
Javascript 

// 2nd method 
const obj = {};

obj[key1] = value1;
