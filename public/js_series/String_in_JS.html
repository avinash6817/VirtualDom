String in Javascript

# Strings are useful for holding data that can be represented in text form. Inshort Whenever we want to use text format we use string(builtin primitive data types in js).

# We use single('') or double quotes("") to represent string.

# Well string is a primitive datatypes But strings can also be defined as objects with the new keyword.

let firstName = new String("avinash");

# But we dont often use new keyword to create strings as objects. It slows down execution speed.The new keyword complicates the code. This can produce some unexpected results, you can visit the W3schools website for deeper dive.

Javascript 

let x = "avinash";
let y = new String("avinash");

console.log(x == y);  // true
console.log(x === y); // false

NOTE - == means value are equal, === means value including the datatypes should also be same. As we understood in last para that one is primitive datatypes (string) and other is objects. you can check it by typeof method.

# String indexing start start from 0, we use length properties to know about the actual length of string.

# To know about the last index we use String.length - 1.

String method & properties
Javascript 

const firstName = 'Avinash';
const lastName = 'Kumar';
const age = 20;

let demo;

// we will use this demo below for many times

concatenation

Javascript 

// concatenation 

demo = firstName + lastName;
console.log(demo); // AvinashKumar

demo = firstName + ' ' + lastName;
console.log(demo); // Avinash Kumar

append

Javascript 

// append(adding) 

demo = 'Avinash';
demo += 'Kumar'; // demo = demo + 'Kumar';

console.log(demo); // Avinash Kumar
Javascript 

// old method 

demo = 'Hello my name is ' + firstName + ' and I am' + age;

console.log(demo); // Hello my name is Avinash and I am 20

escaping

Javascript 

// escaping 

demo = 'That\'s awesome , i can\'t wait';

console.log(demo); // That's awesome , i can't wait

Javascript 

// indexof()

demo = firstName.indexOf('n');

.lastIndexof()

Javascript 

// lastindexof()

let quote = 'hello from the hello world program';
let quoteLetter1 = quote.indexOf('o');
let quoteLetter2 = quote.lastIndexOf('o');
let quote1 = quote.indexOf('hello');
let quote2 = quote.lastIndexOf('hello');
let quote3 = quote.lastIndexOf('code');


console.log(quoteLetter1, quoteLetter2); // 4 29
console.log(quote1, quote2, quote3); // 15 0 -1

.charat()

Javascript 

// charat
demo = firstName.charAt(1);

console.log(demo); // v
Javascript 

// getLastChar

demo = firstName.charAt(firstName.length - 1);

console.log(demo); // h

.substring()

Javascript 

// substring

demo = firstName.substring(0, 4);

console.log(demo); // Avin

.slice()

Javascript 

// slice()

demo = firstName.slice(0, 4);
console.log(demo); // Avin

demo = firstName.slice(-2);
console.log(demo); // sh

.split()

Javascript 

// split() 

const str = 'Hello My name is Avinash';
const tags = 'web development, web design, programming';

demo = str.split(' ');
console.log(demo); // ['Hello', 'My', 'name', 'is', 'Avinash']

demo = tags.split(',');
console.log(demo); // ['web development', ' web design', ' programming']

.replace()

Javascript 

// replace()

const myString = 'oh shit! Here we go again';
demo = myString.replace('shit', 'wow');

console.log(demo);  // oh wow! Here we go again

.includes()

Javascript 

// includes

demo = firstName.includes('a');

console.log(demo);  // true

.trim()

Javascript 

// trim() 

const whiteSpaceString = '   oh shit! Here we go again  '; 
console.log(whiteSpaceString.length); // 30

demo = whiteSpaceString.trim();
console.log(demo.length); // 25 
console.log(demo);        // oh shit! Here we go again

.match()

Javascript 

// match() 

const broCode = 'Do you got it right, yeah i got you';

demo = broCode.match('got');
console.log(demo);  // ['got', index: 7]

// To want more than one match we use globally g prefix 

demo = broCode.match(/got/g); 
console.log(demo);  // ['got', 'got']
