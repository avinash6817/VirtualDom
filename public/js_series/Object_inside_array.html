Object inside Array

# Object inside Array is very useful in real world Applications .

# const users = [ {}, {}, {} ] ;

Javascript 

const users = [{
    userId: 1,
    firstName: 'Avinash',
    gender: 'male'
},
{
    userId: 2,
    firstName: 'Sumit',
    gender: 'male'
},
{
    userId: 3,
    firstName: 'Sonam',
    gender: 'female'
}
];

console.log(users);
// [{...}, {...}, {...}]
Iterating over Array inside objects
Javascript 

for (let user of users) {
    console.log(user);
}

// {userId:1, firstName: 'Avinash', gender: 'male'} 
// {userId:2, firstName: 'Sumit', gender: 'male'} 
// {userId:3, firstName: 'Sonam', gender: 'female'}
Javascript 

// particularly access 

for (let user of users) {
    console.log(user.firstName);
}

// Avinash 
// Sumit 
// Sonam
