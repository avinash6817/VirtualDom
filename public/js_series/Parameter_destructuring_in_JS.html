Parameter Destructuring in Javascript

Javascript 

const person = {
    firstName: 'Avinash',
    gender:'male',
    age:20
}
Javascript 

function withoutDestructuring(obj) {
    console.log(obj.firstName);
    console.log(obj.gender);
}

withoutDestructuring(person);
// Avinash
// male

# used in Object, React

Javascript 

function withDestructuring({firstName,gender,age}) {
    console.log(firstName);
    console.log(gender);
    console.log(age);
}

withDestructuring(person);
// Avinash
// male
// 20
